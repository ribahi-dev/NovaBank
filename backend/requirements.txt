# requirements.txt — dépendances du backend, versions épinglées (==).
#
# Pourquoi épingler : le trinôme, la CI et la machine de démo doivent
# installer EXACTEMENT les mêmes versions. Sans ==, un `pip install` fait
# six semaines plus tard peut tirer une version qui casse tout la veille
# de la soutenance ("ça marche chez moi" garanti).
#
# Règle appliquée ici : on épingle les versions réellement INSTALLÉES ET
# TESTÉES sur Python 3.14 (les anciennes versions du tutoriel n'avaient
# pas de binaires compilés pour 3.14 — leçon retenue).
# =============================================================================

uvicorn[standard]==0.50.2       # serveur ASGI qui exécute FastAPI

# --- Base de données ---

alembic==1.18.5                 # migrations versionnées du schéma (le "git de la base")
